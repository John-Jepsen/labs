    Example Configuration

Sales by Category =
CALCULATE(
    SUM(Sales[Amount]),
    ALLSELECTED(Products[Category])
)

// Measure for trend
Sales Trend =
CALCULATE(
    [Total Sales],
    DATESYTD('Date'[Date])
)
